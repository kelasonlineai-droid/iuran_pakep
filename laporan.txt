    <script>
        const pesanGagal = document.getElementById('pesanGagal');
        const teksError = document.getElementById('teksError');
        const btnCari = document.getElementById('btnCari');
        const btnSemua = document.getElementById('btnSemua');
        const tglAwal = document.getElementById('tglAwal');
        const tglAkhir = document.getElementById('tglAkhir');
        const wadahTabel = document.getElementById('wadahTabel');
        const isiTabelData = document.getElementById('isiTabelData');
        const wadahPagination = document.getElementById('wadahPagination');
        const btnFullscreen = document.getElementById('btnFullscreen');
        const iconFullscreen = btnFullscreen.querySelector('i');

        // ✅ FITUR LAYAR PENUH
        btnFullscreen.addEventListener('click', function() {
            if (!document.fullscreenElement) {
                document.documentElement.requestFullscreen();
                iconFullscreen.className = 'bi bi-fullscreen-exit';
                btnFullscreen.classList.add('active');
            } else {
                if (document.exitFullscreen) {
                    document.exitFullscreen();
                    iconFullscreen.className = 'bi bi-arrows-fullscreen';
                    btnFullscreen.classList.remove('active');
                }
            }
        });

        // ✅ KONFIGURASI: 25 DATA PER HALAMAN
        const ITEM_PER_PAGE = 25;
        let dataGlobal = [];
        let halamanSaatIni = 1;

        btnSemua.addEventListener('click', function() {
            isiTabelData.innerHTML = '<tr><td colspan="4" class="text-center text-muted py-3">⏳ Memuat semua data...</td></tr>';
            wadahPagination.innerHTML = '';
            
            google.script.run
                .withSuccessHandler(function(data) {
                    dataGlobal = data;
                    halamanSaatIni = 1;
                    renderHalaman();
                })
                .withFailureHandler(function(err) {
                    pesanGagal.classList.remove('d-none');
                    teksError.textContent = "❌ Gagal memuat: " + err;
                    setTimeout(() => pesanGagal.classList.add('d-none'), 3000);
                })
                .ambilSemuaDataIuran();
        });

        btnCari.addEventListener('click', function() {
            if(!tglAwal.value || !tglAkhir.value){
                pesanGagal.classList.remove('d-none');
                teksError.textContent = "⚠️ Harap isi tanggal awal dan akhir!";
                setTimeout(() => pesanGagal.classList.add('d-none'), 3000);
                return;
            }
            if(new Date(tglAwal.value) > new Date(tglAkhir.value)){
                pesanGagal.classList.remove('d-none');
                teksError.textContent = "⚠️ Tanggal awal tidak boleh lebih besar dari akhir!";
                setTimeout(() => pesanGagal.classList.add('d-none'), 3000);
                return;
            }

            isiTabelData.innerHTML = '<tr><td colspan="4" class="text-center text-muted py-3">⏳ Mencari data...</td></tr>';
            wadahPagination.innerHTML = '';
            
            google.script.run
                .withSuccessHandler(function(data) {
                    dataGlobal = data;
                    halamanSaatIni = 1;
                    renderHalaman();
                })
                .withFailureHandler(function(err) {
                    pesanGagal.classList.remove('d-none');
                    teksError.textContent = "❌ Gagal mencari: " + err;
                    setTimeout(() => pesanGagal.classList.add('d-none'), 3000);
                })
                .ambilDataByTanggal(tglAwal.value, tglAkhir.value);
        });

        function renderHalaman() {
            isiTabelData.innerHTML = "";

            if(dataGlobal.length <= 1){
                isiTabelData.innerHTML = '<tr><td colspan="4" class="text-center text-danger py-3">❌ Tidak ada data ditemukan</td></tr>';
                wadahPagination.innerHTML = '';
                return;
            }

            const totalData = dataGlobal.length - 1;
            const totalHalaman = Math.ceil(totalData / ITEM_PER_PAGE);

            const mulai = (halamanSaatIni - 1) * ITEM_PER_PAGE + 1;
            const selesai = mulai + ITEM_PER_PAGE;
            const dataDitampilkan = dataGlobal.slice(mulai, selesai);

            dataDitampilkan.forEach(barisData => {
                let baris = document.createElement('tr');
                barisData.forEach( isi => {
                    let kolom = document.createElement('td');
                    kolom.textContent = isi;
                    baris.appendChild(kolom);
                });
                isiTabelData.appendChild(baris);
            });

            buatNavigasiHalaman(totalHalaman);
        }

        function buatNavigasiHalaman(totalHalaman) {
            wadahPagination.innerHTML = '';

            const btnPrev = document.createElement('button');
            btnPrev.className = 'page-btn page-nav';
            btnPrev.innerHTML = '<i class="bi bi-chevron-left"></i>';
            btnPrev.disabled = (halamanSaatIni === 1);
            btnPrev.onclick = () => { halamanSaatIni--; renderHalaman(); };
            wadahPagination.appendChild(btnPrev);

            for(let i = 1; i <= totalHalaman; i++){
                const btnNomor = document.createElement('button');
                btnNomor.className = 'page-btn ' + (i === halamanSaatIni ? 'active' : '');
                btnNomor.textContent = i;
                btnNomor.onclick = () => { halamanSaatIni = i; renderHalaman(); };
                wadahPagination.appendChild(btnNomor);
            }

            const btnNext = document.createElement('button');
            btnNext.className = 'page-btn page-nav';
            btnNext.innerHTML = '<i class="bi bi-chevron-right"></i>';
            btnNext.disabled = (halamanSaatIni === totalHalaman);
            btnNext.onclick = () => { halamanSaatIni++; renderHalaman(); };
            wadahPagination.appendChild(btnNext);
        }

    </script>
