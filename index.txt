    <script>
        const form = document.getElementById('formIuran');
        const pesanSukses = document.getElementById('pesanSukses');
        const pesanGagal = document.getElementById('pesanGagal');
        const teksSukses = document.getElementById('teksSukses');
        const teksError = document.getElementById('teksError');
        const selectNama = document.getElementById('namaWarga');
        const inputNominal = document.getElementById('nominalIuran');
        const inputNilaiAsli = document.getElementById('nilaiAsli');
        const btnFullscreen = document.getElementById('btnFullscreen');
        const iconFullscreen = btnFullscreen.querySelector('i');

        // ==============================================
        // ✅FITUR LAYAR PENUH
        // ==============================================
        btnFullscreen.addEventListener('click', function() {
            if (!document.fullscreenElement) {
                document.documentElement.requestFullscreen();
                iconFullscreen.className = 'bi bi-fullscreen-exit';
                btnFullscreen.classList.add('active');
            } else {
                if (document.exitFullscreen) {
                    document.exitFullscreen();
                    iconFullscreen.className = 'bi bi-arrows-fullscreen';
                    btnFullscreen.classList.remove('active');
                }
            }
        });

        // ==============================================
        // ✅ FORMAT RUPIAH OTOMATIS
        // ==============================================
        inputNominal.addEventListener('input', function () {
            let angka = this.value.replace(/\D/g, '');
            inputNilaiAsli.value = angka;
            if (angka) this.value = new Intl.NumberFormat('id-ID').format(angka);
        });

        // ==============================================
        // ✅ MUAT DAFTAR NAMA WARGA
        // ==============================================
        function tampilkanDaftarNama() {
            google.script.run
                .withSuccessHandler(function(hasil) {
                    selectNama.innerHTML = '';
                    let opsiAwal = document.createElement('option');
                    opsiAwal.value = "";
                    opsiAwal.textContent = "-- Pilih Nama --";
                    opsiAwal.disabled = true;
                    opsiAwal.selected = true;
                    selectNama.appendChild(opsiAwal);

                    hasil.forEach(function(item) {
                        let opsi = document.createElement('option');
                        opsi.value = item.id;
                        opsi.textContent = item.nama;
                        selectNama.appendChild(opsi);
                    });
                })
                .withFailureHandler(function(err) {
                    selectNama.innerHTML = '<option value="" disabled>❌ ' + err + '</option>';
                })
                .getDaftarWarga();
        }
        window.onload = tampilkanDaftarNama;

        // ==============================================
        // ✅ SIMPAN DATA
        // ==============================================
        form.addEventListener('submit', function(e) {
            e.preventDefault();
            let idTerpilih = selectNama.value;
            let namaTerpilih = selectNama.options[selectNama.selectedIndex].text;
            let jumlah = inputNilaiAsli.value;

            if (!idTerpilih || !jumlah || jumlah === "0") {
                pesanGagal.classList.remove('d-none');
                teksError.textContent = "⚠️ Pilih nama & isi jumlah dulu!";
                setTimeout(() => pesanGagal.classList.add('d-none'), 3000);
                return;
            }

            pesanSukses.classList.add('d-none');
            pesanGagal.classList.add('d-none');

            google.script.run
                .withSuccessHandler(() => {
                    pesanSukses.classList.remove('d-none');
                    teksSukses.textContent = "✅ Data berhasil disimpan!";
                    form.reset();
                    inputNilaiAsli.value = "";
                    setTimeout(() => pesanSukses.classList.add('d-none'), 2000);
                })
                .withFailureHandler((err) => {
                    pesanGagal.classList.remove('d-none');
                    teksError.textContent = "❌ Gagal: " + err;
                    setTimeout(() => pesanGagal.classList.add('d-none'), 4000);
                })
                .simpanDataIuran(idTerpilih, namaTerpilih, jumlah);
        });

    </script>
